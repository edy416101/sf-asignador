En index.html, reemplaza: GOOGLE_CLIENT_ID = "TU_ID_AQUI" // // ══════════════════════════════════════════════════════════════════════════ // ── Endpoint principal (recibe el POST de la web app) ──────────────────── function doPost(e) { var output = ContentService.createTextOutput(); output.setMimeType(ContentService.MimeType.JSON); try { // NOTA: El frontend envía sin Content-Type para evitar el preflight CORS. // El body llega igualmente como JSON en e.postData.contents. var payload = JSON.parse(e.postData.contents); var resultado = procesarAsignacion(payload); output.setContent(JSON.stringify(resultado)); } catch (err) { output.setContent(JSON.stringify({ error: err.message, stack: err.stack })); } return output; } // ── Health check (GET) ─────────────────────────────────────────────────── function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ status: "OK", version: "2.0", service: "SF Asignador" })) .setMimeType(ContentService.MimeType.JSON); } // ══════════════════════════════════════════════════════════════════════════ // LÓGICA PRINCIPAL — Alineada 100% con el Apps Script original de Sheets // ══════════════════════════════════════════════════════════════════════════ function procesarAsignacion(params) { var sessionId = params.sid; var instanceUrl = params.instanceUrl; // siempre "https://docplanner.my.salesforce.com" var dias = params.dias; var businessLine = params.businessLine; var reps = params.reps; // [{email, cuota}] var idsContactos = params.ids; // ["001xxx", "001yyy", ...] // Opciones comunes para todas las llamadas a Salesforce var opciones = { "method": "GET", "headers": { "Authorization": "Bearer " + sessionId, "Content-Type": "application/json" }, "muteHttpExceptions": true }; // ── PASO 1: Buscar Owner IDs en Salesforce ──────────────────────────── // Idéntico al original: busca por Email Y Username para máxima cobertura var emailsList = reps.map(function(r) { return "'" + r.email + "'"; }); var soqlUsers = "SELECT Id, Email, Username FROM User " + "WHERE Email IN (" + emailsList.join(",") + ") " + "OR Username IN (" + emailsList.join(",") + ")"; var urlUsers = instanceUrl + "/services/data/v57.0/query/?q=" + encodeURIComponent(soqlUsers); var resUsers = UrlFetchApp.fetch(urlUsers, opciones); var jsonUsers = JSON.parse(resUsers.getContentText()); if (!jsonUsers.records) { throw new Error("Salesforce no devolvió registros de usuarios. Verifica el SID. Respuesta: " + JSON.stringify(jsonUsers).slice(0, 200)); } // Construir mapa email/username → Id (mismo comportamiento que el original) var mapUserIds = {}; jsonUsers.records.forEach(function(u) { if (u.Email) mapUserIds[u.Email.toLowerCase().trim()] = u.Id; if (u.Username) mapUserIds[u.Username.toLowerCase().trim()] = u.Id; }); // ── PASO 2: Extraer contactos en bloques (igual que el original) ────── // El original usa bloques de 40 para no superar límites de SOQL var poolContactos = []; var tamanoBloque = 40; for (var i = 0; i < idsContactos.length; i += tamanoBloque) { var bloque = idsContactos.slice(i, i + tamanoBloque); var idsFormateados = bloque.map(function(id) { return "'" + id + "'"; }).join(","); // SOQL idéntico al código original de Sheets var soqlContactos = "SELECT Id, LastSource_SO__c, LastSourceAt__c, Account.Name, LastName, Email, Source_SO__c, SourceSOAt__c " + "FROM Contact " + "WHERE Id IN (" + idsFormateados + ") " + "AND Id NOT IN (" + "SELECT Contact__c FROM Opportunity " + "WHERE IsClosed = False OR (IsClosed = True AND CloseDate = LAST_N_DAYS:" + dias + ")" + ") " + "AND Id NOT IN (" + "SELECT Contact__c FROM Lead " + "WHERE Status IN ('New', 'Assigned', 'Qualifying') OR lead_closed_date__c = LAST_N_DAYS:" + dias + ")"; var urlContactos = instanceUrl + "/services/data/v57.0/query/?q=" + encodeURIComponent(soqlContactos); try { var resContactos = UrlFetchApp.fetch(urlContactos, opciones); var jsonContactos = JSON.parse(resContactos.getContentText()); if (jsonContactos.records) { jsonContactos.records.forEach(function(c) { var accName = c.Account ? c.Account.Name : "Sin Cuenta"; poolContactos.push([ c.Id, c.LastSource_SO__c || "", c.LastSourceAt__c || "", accName, c.LastName || "", c.Email || "", c.Source_SO__c || "", c.SourceSOAt__c || "" ]); }); } } catch (err) { Logger.log("Error en bloque de contactos (índice " + i + "): " + err.message); } } // ── PASO 3: Distribuir contactos entre reps ─────────────────────────── // Lógica idéntica al original: puntero secuencial, cuota por rep var filasFinal = []; var puntero = 0; // Valores fijos exactos del código original var valoresFijos = ["New Business", "DOCTOR", "Mexico", "Assigned", "FALSE"]; reps.forEach(function(rep) { var email = rep.email.toLowerCase().trim(); var ownerId = mapUserIds[email] || "ID NO ENCONTRADO"; var cuota = rep.cuota || 0; for (var k = 0; k < cuota; k++) { if (puntero >= poolContactos.length) break; var contacto = poolContactos[puntero++]; var filaFinal = contacto .concat([ownerId, businessLine]) .concat(valoresFijos); filasFinal.push(filaFinal); } }); // ── RESPUESTA ───────────────────────────────────────────────────────── // rows: array de arrays (15 columnas por fila), equivale al "pre template" // poolTotal: cuántos contactos pasaron los filtros de exclusión // assigned: cuántos se asignaron efectivamente return { rows: filasFinal, poolTotal: poolContactos.length, assigned: filasFinal.length }; }
